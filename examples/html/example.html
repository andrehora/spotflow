14 def parseparam(s):
15     s = ';' + str(s)
16     plist = []
17     while s[:1] == ';':
18         s = s[1:]
19         end = s.find(';')
20         while end > 0 and (s.count('"', 0, end) - s.count('\\"', 0, end)) % 2:
21             end = s.find(';', end + 1)
22         if end < 0:
23             end = len(s)
24         f = s[:end]
25         if '=' in f:
26             i = f.index('=')
27             f = f[:i].strip().lower() + '=' + f[i+1:].strip()
28         plist.append(f.strip())
29         s = s[end:]
30     return plist
